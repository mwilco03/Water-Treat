#
# sysconf.txt - Raspberry Pi first-boot configuration
#
# What: Pre-bake configuration for Water-Treat RTU
# Why: Zero-touch provisioning without cloud-init dependency
#
# Usage: Place on boot partition (/boot/firmware/)
#        Alongside firstrun.sh
#
# Compatible with: Raspberry Pi OS, DietPi, Armbian
#

# ------------------------------------------------------------------------------
# Hostname (temporary - firstrun.sh sets rtu-XXXX from MAC)
# ------------------------------------------------------------------------------
hostname=rtu

# ------------------------------------------------------------------------------
# User Configuration
# ------------------------------------------------------------------------------
user_name=admin
user_password=H2OhYeah!
user_groups=sudo,gpio,i2c,spi,dialout

# ------------------------------------------------------------------------------
# SSH Configuration
# ------------------------------------------------------------------------------
ssh_enabled=1
ssh_pwauth=1

# ------------------------------------------------------------------------------
# Locale and Keyboard
# ------------------------------------------------------------------------------
locale=en_US.UTF-8
keyboard_layout=us
timezone=America/New_York

# ------------------------------------------------------------------------------
# First Boot Script
# ------------------------------------------------------------------------------
# Points to the provisioning script on the boot partition
firstrun_script=/boot/firmware/firstrun.sh
